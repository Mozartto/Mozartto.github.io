关系型数据库技术

Posted on 2020-06-30 Edited on 2020-09-01 In computer Changyan:
Symbols count in article: 22k Reading time ≈ 20 mins.
关系型数据库，是指采用了关系模型来组织数据的数据库，其以行和列的形式存储数据，以便于用户理解，关系型数据库这一系列的行和列被称为表，一组表组成了数据库。用户通过查询来检索数据库中的数据，而查询是一个用于限定数据库中某些区域的执行代码。关系模型可以简单理解为二维表格模型，而一个关系型数据库就是由二维表及其之间的关系组成的一个数据组织。

这里以MySQL为例。

MySQL数据库

MySQL基本上是企业中最常用的关系型数据库了，关于MySQL的使用和可视化图形界面工具Navicat的安装这里不详细展开。(不过默默提醒一句，Navicat真的太贵了，三个月1699￥，建议想用的但是还没有参加工作的同学先淘宝，不过是真的好用，比SQLYog好用得不是一个级别)

非外键约束：

为防止不符合规范的数据存入数据库，在用户对数据进行插入、修改、删除等操作时，MySQL 提供了一种机制来检查数据库中的数据是否满足规定的条件，以保证数据库中数据的准确性和一致性，这种机制就是完整性约束。

MySQL中主要支持一下几种完整性约束，如表所示，其中check约束是MySQL8中提供的支持

比如上表中学生学号重复了，这是不对的，所以此时我们需要加入约束

主键：不能为空 + 唯一 ，比如上表中学号就应该是主键

主键的作用：通过主键，可以识别到唯一的一条记录

如果主键是整数类型，那么需要自增

非外键约束的类型：

CREATE TABLE student (
	num INT ( 6 ) primary key auto_increment, -- 主键约束，且自增
	NAME VARCHAR ( 10 ) not null, -- 非空约束
	sex CHAR ( 1 ) default '男' /* 默认约束*/  check(sex = '男' || '女'), -- 检查约束
	age INT ( 3 ) check(age >= 18 and age <= 50),
	enterdate date,
classname VARCHAR ( 10 ),
email VARCHAR ( 15 ) unique -- 唯一约束
); 

-- 添加数据	
insert into student values(1,'张三','男',18,now(),'软件工程01班','zs@163.com');
SELECT * from student;
insert into student values(2,'张三','男',88,now(),'软件工程01班','zs@1163.com');
-- 下面这个年龄超出范围，添加数据时报错

-- 插入部分数据
insert into student (name, enterdate) values ('feife','2018-04-23');
-- 添加null或default的话主键一样会自增(其它部分数据有效)，添加时主键部分数据添加成功，其它部分数据导致添加失败时，主键一样会自增
-- 主键不要求连号，唯一且非空即可

约束从作用上可以分为两类：

  1. 表级约束：可以约束表中任意一个或多个字段。与列定义相互独立，不包含在列定义中。与定义用 ‘,’ 分隔；必须指出要约束的名称；
  2. 列级约束：包含在列定义中，直接跟在该列的其它定义之后，用空格分隔，不必指定列名；
create table student(
	num int(6) auto_increment, 
	name varchar(10) not null, 
	sex char(1) default '男',
	age int(3),
	enterdate date,
	classname varchar(10), 
	email varchar(15),
	
	constraint pk_stu primary key(num),-- 这就是一个约束，pk_stu是约束的名字，我们添加的是一个主键约束,后面可以指定是给哪个列添加的约束
	-- 如果想要自增，auto_increment只能添加到列的后面去，不能添加在单独写出来的主键约束后面
	
	constraint ck_stu_sex check (sex = '男' || sex = '女'),
	-- 添加检查约束时记得标明是对哪个列信息的检查约束，以免报错的时候分不清楚
	
	constraint ck_stu_age check (age >= 18 and age <= 50),
	
	constraint uq_stu_email unique (email)
	
	-- 非空约束和默认约束不能加表级约束，只能添加到列级约束里面
	);
	insert into student values(1,'张三','男',18,now(),'软件工程01班','zs@163.com');
	insert into student values(5,'王五','男',18,now(),'软件工程01班','ww@163.com');
	
	select * from student;
	-- 注意，主键虽然是默认自增，但是你也可以1，2，5的这样去插入不连续的主键，这个时候系统以你写入的为主，如果你没有写入或写入null，则其会自动自增

上面讲过的所有约束都是非外键约束，下面来说一说外键约束，外键约束指的是表中的某一个字段依赖于另一张表中某一字段的值。

-- 先创建父表：班级表
create table class(
	classnum int(4) primary key auto_increment,
	cname varchar(10) not null,
	room char(4)
	);
	
-- 添加班级数据
insert into class values(null,'java001','r003');
insert into class values(null,'java002','r416');
insert into class values(null,'java001','r103');

desc class;
select * from class;

-- 一个一个添加太麻烦了，可以一次性添加多个信息：
insert into class values (null,'java001','r003'),(null,'java002','r416'),(null,'java001','r103');

-- 创建子表(学生表)。在这里班级表是父表，学生表是子表
-- 创建学生表：
create table student(
	num int(6) primary key auto_increment, 
	name varchar(10) not null, 
	classname int(4) -- 取值参考class表中的cno字段，但是不要求字段名字完全重复，但是类型和长度定义要求是要完全相同的 
);

-- 添加学生信息：
insert into student values (null,'张三',1),(null,'李四',1),(null,'王五',2);
select * from student;
update student set classname = 2 where num = 3;

-- 出现问题：
-- 1.添加一个学生对应的班级编号为4(但实际上没有4班）

-- 2.删除班级2，不能说删就删，因为里面有学生。删除后学生表中的学生还是属于2班
delete from class where classnum = 2;
-- 出现问题的原因：
-- 因为你现在的外键约束根本没用语法添加进去，现在只是逻辑上认为班级编号是外键，但是没有从语法上定义

-- 解决办法：添加外键约束
-- 注意：外键约束只有表级约束，没有列级约束
drop table student;

create table student(
	num int(6) primary key auto_increment, 
	name varchar(10) not null, 
	classnum int(4), -- 取值参考class表中的cno字段，但是不要求字段名字完全重复，但是类型和长度定义要求是要完全相同的 
	
	constraint fk_stu_classnum foreign key (classnum) references class (classnum)
);


-- 在创建表以后添加外键约束：
create table student(
	num int(6) primary key auto_increment, 
	name varchar(10) not null, 
	classname int(4) -- 取值参考class表中的cno字段，但是不要求字段名字完全重复，但是类型和长度定义要求是要完全相同的 
);

alter table student add constraint 	fk_stu_classnum foreign key (classnum) references class (classnum);

insert into student values (null,'张三',1),(null,'李四',1),(null,'王五',2);
-- 这时添加2班的学生会发现添加失败
select * from student;
update student set classname = 2 where num = 3;

-- 此时删除已有学生的班级也会发现删除失败：
delete from class where classnum = 1;

注意：删除含有外键约束的表时，先删主表，再删从表

外键策略：

create table class(
	classnum int(4) primary key auto_increment,
	cname varchar(10) not null,
	room char(4)
	);
insert into class values (null,'java001','r003'),(null,'java002','r416'),(null,'java001','r103');


create table student(
	num int(6) primary key auto_increment, 
	name varchar(10) not null, 
	classname int(4) -- 取值参考class表中的cno字段，但是不要求字段名字完全重复，但是类型和长度定义要求是要完全相同的 
	constraint fk_stu_classnum foreign key (classnum) references class (classnum)
);

insert into student values (null,'张三',1),(null,'李四',1),(null,'王五',2);

select * from student;
select * from class;

-- 删除班级2，如果直接删除的话肯定不行，因为有外键约束
-- 加入外键策略：
-- 策略1：no action 不允许操作
-- 不能直接删，可以间接删，通过操作sql来完成
-- 先把班级2的学生对应的班级改为3班或改为null，然后再删除班级2
update student set classnum = 3 where classnum = 2;
delete from class where classnum = 2; 

-- 策略2：使用cascate 级联操作：(在操作主表时，它会影响从表的外键信息)
-- 先删除之前的外键约束，然后重新添加外键约束
-- 先试试更新操作：
update class set classnum = 2 where classnum = 5;

alter table student drop foreign key fk_stu_classnum;

alter table student add constraint fk_stu_classnum foreign key (classnum) references class (classnum) on update cascade on delete cascade;

-- 此时你更新class表里面的班级，学生表里面对应班级的学生的班级信息也会自动更改
-- 级联操作不要轻易去用，因为如果有误操作，可能会对很多数据造成影响

-- 策略3：set null 置空操作
alter table student add constraint fk_stu_classnum foreign key (classnum) references class (classnum) on update set null on delete set null;

-- 级联操作和置空操作也可以混着写

-- 应用场合：
-- 1.朋友圈删除，点赞留言都删除：级联操作
-- 2.解散班级，但学生不能删除：置空操作

DDL和DML的补充：

-- 快速添加和已有表结构相同的表
create table student2
as 
select * from student;
-- 结构和数据都和student表是一致的

-- 快速添加结构一致但是没有数据的表：
create table student3
as 
select * from student where 1 = 2;
-- 1=2永远不满足，数据添加不了，但是结构会添加进来

-- 快速添加部分列和部分数据：
create table student4
as 
select name,classnum from student where num = 4;

-- 删除操作：
delete from student;
truncate table student;
-- delete和truncate都是删除的操作，delete属于DML操作，delete是一条一条的删数据，可以加where
-- truncate是保留表的结构，直接删除整张表然后创建一张空表
-- delete操作可以回滚，truncate操作会导致隐式提交，因此不能回滚
-- delete操作会显示影响行数，truncate操作显示的影响行数为0
-- delete删除数据后数据还是会从之前的最大值加1开始自增，truncate删除数据后数据从1开始重新自增

说几个简单的SQL查询语句的用法：

-- 对emp进行查询
select * from emp;

select empno, emname, sal from emp;

-- 起别名(中间可以加as，也可以省略,别名可以加单引号或双引号，别名中间有空格或特殊符号时时必须加引号)
select empno 员工编号, emname 员工名字, sal 员工工资 from emp;

-- 在查询过程中加入算数运算符
select empno, emname, sal, sal+1000 '涨薪后', deptno from emp where sal < 7500;
select empno, emname, sal, comm, sal + comm from emp;
-- 注意：sal和comm中只要有一个为mull，则相加结果为null

-- 去重操作,在你选的选项前面加一个distinct
select job from emp;
select distinct job from emp;

-- 排序: (默认情况下是升序排列)
select * from emp order by sal;
select * from emp order by sal asc; -- 指定升序排列
select * from emp order by sal desc; -- 指定降序排列

-- 在工资相同时，按部门降序排序
select * from emp order by sal asc, deptno desc;
select * from emp where  binary deptno = 10; -- 加binary代表区分大小写
select * from emp where sal between 0 and 3000; -- 工资在1500和3000之间
select * from emp where deptno in (10,20); -- 只能是10或者20，不是区间噢
select * from emp where job in ('clerk1','clerk2','clerk12'); -- 和上面同理

-- where子句 + 模糊查询：

使用函数：

单行函数：

单行函数：单行函数是指对每一条输入值进行计算，并得到相应的计算结果，然后返回给用户，也就是说，每条记录作为一个输入参数，经过函数计算得到相应结果。

常用的单行函数主要包括字符串函数、数值函数、日期与时间函数、流程函数以及其他函数。

子查询

-- 引入子查询：
-- 查询所有比“CLERK"工资高的员工的信息

select sal from emp where sal > 6800;

事务及其特征：

事务（transaction）是用来维护数据库完整性的，它能够保证一系列的MySQL操作要么全部执行，要么全不执行

事务的四大特性（ACID)：原子性、一致性、隔离性、持久性

-- sql展示：通过事务确保转账安全
-- 创建一个账户表：
create table account(
	id int primary key auto_increment,
	uname varchar(10) not null,
	balance double
);
select * from account;

-- 在表中插入数据：
insert into account values(null, '丽丽', 2000), (null, '小刚',2000);

-- 让丽丽给小刚转200块钱：
update account set balance = balance - 200 where id = 1;
update account set balance = balance + 200 where id = 2;
-- 默认一个DML语句是一个事务
-- 所以上面的操作执行了两个事务

-- 必须让上面的两个操作控制在一个事务中：

-- 手动开启事务：
start transaction;

update account set balance = balance - 200 where id = 1;
update account set balance = balance + 200 where id = 2;

-- 手动回滚：刚才执行的操作全部取消：
rollback;

-- 提交
commit;

-- 在回滚和提交之前，数据库中的数据都是操作的缓存中的数据，而不是数据库的真实数据
-- 记得在使用回滚和提交之前先手动开启事务，不然rollback和commit都没用...

事务并发问题：

可能发生的并发问题一共有下面三种：脏读（Dirty read）、不可重复读（解决办法是锁住一行数据）、幻读（phantom read）–解决办法是锁住表

事物的隔离级别

最后一个序列化可以解决所有问题，但是速度慢，我们要权衡各种利弊选择不同的隔离级别。一般我们都用重复读的隔离级别。MySQL用的默认级别就是重复读

-- 查看事务的默认隔离级别：
select @@transaction_isolation;

-- 设置事务的隔离级别，设置当前会话的隔离级别：
set session transaction isolation level read uncommitted;
set session transaction isolation level read read committed;
set session transaction isolation level read repeatable read;
set session transaction isolation level read serializable;

		select * from emp;
	else select * from emp where emname like concat("%", name, "%"); 
	-- 在MySQL里面拼接字符串需要使用字符串拼接函数concat
	end if;
end;

-- 删除存储过程：
drop procedure myprocedure01;

-- 调用存储过程：
call myprocedure01(null);
	
call myprocedure01('k');


-- 定义一个有返回值的存储过程：

create procedure myprocedure02(in name varchar(10), out num int(3))
 -- 返回值写在参数列表中，用in和out区分参数和返回值
begin
	if name is null or name = '' then
		select * from emp;
	else select * from emp where emname like concat("%", name, "%"); 
	-- 在MySQL里面拼接字符串需要使用字符串拼接函数concat
	end if;
	select found_rows() into num;
	-- found_rows()是MySQL中定义的一个函数，作用是返回查询结果的条数
end;

-- 调用存储过程：

call myprocedure02('i', @num); -- 用num接收返回结果
select @num;
