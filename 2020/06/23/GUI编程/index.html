                System.exit(0);

        });

    4.事件监听

    事件监听：当某个事情发生的时候，干什么？

    对于点一个按钮然后就想在控制台输出内容的可以用frame.addActionListener对于点击×就可以关闭窗口的监听可以用frame.addWindowListener

    public class TestActionEvent {
    public static void main(String[] args) {
        //按下按钮的时候出发一些事件
        Frame frame  = new Frame();

        Button button = new Button();

        //因为addActionListener需要一个ActionListener，所以我们需要构造一个ActionListener
        //怎么构造呢？接口就写实现类
        MyActionListener myActionListener = new MyActionListener();
        button.addActionListener(myActionListener);

        frame.add(button, BorderLayout.CENTER);
        frame.pack();
        frame.setVisible(true);
        windowClose(frame);
    }

    //关闭窗体的事件
    private static void windowClose(Frame frame){
        frame.addWindowListener(new WindowAdapter() {  
            //这是一个匿名内部类，里面重写了一个方法
            //方法的参数是WindowEvent类
            @Override
            public void windowClosing(WindowEvent e) {
                System.exit(0);
            }
        });
    }
}
//事件监听
class MyActionListener implements ActionListener{  
    @Override
    public void actionPerformed(ActionEvent e){
        System.out.println("aaa");
    }
}

    多个按钮共享一个事件：

    public class TestActionListener02 {

        Frame frame = new Frame("开始，停止");

        //两个按钮共用同一个监听
        //开始    停止
        Button button1 = new Button("start");
        Button button2 = new Button("pause");

        button2.setActionCommand("button-pause");

        MyMonitor myMonitor = new MyMonitor();
        button1.addActionListener(myMonitor);
        button2.addActionListener(myMonitor);
        //两个按钮共用一个监听事件，所以我们现在写一个监听事件

        frame.add(button1,BorderLayout.NORTH);
        frame.add(button2,BorderLayout.SOUTH);
        frame.setVisible(true);
        frame.pack();

        frame.addWindowListener(new WindowAdapter() {
            @Override
            public void windowClosing(WindowEvent e) {
                System.exit(0);
            }
        });
    }
}
class MyMonitor implements ActionListener {
    @Override
    public void actionPerformed(ActionEvent e){
        System.out.println("按钮被点击了：msg"+e.getActionCommand());
        if(e.getActionCommand().equals("start")){
            System.out.println("开始了！");
        }
    }
}

    public Calculator(){

//监听器类
class MyCalculatorListener implements ActionListener {

    //获取三个变量
    private TextField num1, num2, num3;
    public MyCalculatorListener(TextField num1, TextField num2, TextField num3){
        this.num1 = num1;
        this.num2 = num2;
        this.num3 = num3;
    }
    @Override
    public void actionPerformed(ActionEvent e){
        //1.获得加数与被加数
        int n1 = Integer.parseInt(num1.getText());
        int n2 = Integer.parseInt(num2.getText());


        //2.等于时将这两个值进行加法运算后放到第三个框
        num3.setText((n1 + n2)+" ");

        //3.清除前两个框
        num1.setText("");
        num2.setText("");
        }
}

    一般狂神偏好的步骤：

    1. 先把所有组件写出来
    2. 把监听器写出来
    3. 把布局写出来

    自己总结了一个大体思路：

    1. 写一个主方法类，里面调用窗口类
    2. 写一个窗口类，里面有基本的布局以及监听器
    3. 写一个监听器类，并在监听器类中组合窗口类，将窗口类对象传入监听器的抽象方法中，并用button调用这个方法，这样在点击按钮的时候会启动监听，执行这个方法里面的内容

    8 .鼠标监听

    repaint()鼠标每次点击就重画一次

    //测试鼠标监听事件
public class TestMouseListener {
    public static void main(String[] args) {
        new MyFrame("画图");
    }
}
class MyFrame extends Frame {
    //画画需要画笔，需要监听鼠标当前的位置，需要一个集合来存储这个点
    ArrayList points ;

    public MyFrame(String title){
        super(title);
        setBounds(200,200,400,300);
        //我们需要一个东西来存鼠标点击的点
        points = new ArrayList<>();


        this.setVisible(true);

        //鼠标监听器针对当前这个窗口
        this.addMouseListener(new MyMouseListener());
    }
    public void paint(Graphics g) {
        //画画，需要监听鼠标的事件
        Iterator iterator = points.iterator();
        while(iterator.hasNext()){
            Point point = (Point)iterator.next();
            g.setColor(BLUE);
            g.fillOval(point.x, point.y,10,10);
        }
    }

    //添加一个点到界面上
    public void addPaint(Point point){
        points.add(point);
    }

    private class MyMouseListener extends  MouseAdapter {
        //适配器模式不用重写接口中所有的方法，把继承接口改为继承其实现类
        //鼠标，按下，弹起，按住不放
        @Override
        public void mousePressed(MouseEvent e){
            super.mousePressed(e);
            MyFrame myFrame = (MyFrame) e.getSource();
            //这里我们点击的时候就会在界面上产生一个点
            //这个点就是鼠标的点：
            myFrame.addPaint(new Point(e.getX(), e.getY()));

            //每次点击鼠标都需要重画一遍
            myFrame.repaint();//刷新
        }
    }
}

    先有一个窗口，窗口内有一个画笔，然后有一个鼠标监听器（继承了鼠标的适配器），他们三个都是空的。然后我们需要一个缓冲的地方，就是这个鼠标的点应该存在哪里。于是我们就写了一个存点的point集合.这个集合所有人都要用，所以我们把它提到最高。然后这个点要开始放了，我们要把当前鼠标点击的位置给存进来。存到集合里面，然后我们的集合就把它给读取出来。然后我们发现第一次读取它是白的，点了没反应，因为我们点击的时候它已经画完了，我们需要它不断地重新画。

    详细注释版鼠标监听画笔

    //鼠标监听事件
public class TestMouseListener {
    public static void main(String[] args) {
        new MyFrame("画图");
    }
}

//自己的类
class MyFrame extends Frame{
    //画画需要画笔，需要监听鼠标当前的位置，需要一个集合来存储鼠标画出来的点
    ArrayList points;

    public MyFrame(String title){
        super(title);
        setBounds(200,200,400,300);
        this.setVisible(true);
        //存鼠标点的点
        //因为下面的类可能用得到这个集合，所以把它的定义放到最高，然后在这里初始化即可
        points = new ArrayList<>();



        //鼠标监听器，正对这个窗口
        this.addMouseListener(new MyMouseListener());
        //这里面需要传入的是一个MouseListener的实现类对象
        //MouseListener也是一个接口，里面有click，press等等与鼠标监听相关的抽象方法
    }
    @Override
    public void paint(Graphics g){
        //画画，需要监听鼠标的事件
        //这里面画出来的每一个点都需要与鼠标绑定
        //鼠标怎么来了，需要添加一个鼠标的监听器

        //画笔画出来的东西要根据鼠标当前的位置来
        //所以我们现在需要一个东西来存点
        Iterator it = points.iterator();
        while(it.hasNext()){
            Point point = (Point) it.next();
            //ArrayList集合没有指定泛型时默认的泛型是Object型，所以需要强转
            g.setColor(Color.BLUE);
            g.fillOval(point.x, point.y, 10, 10);
            //上面这一步是什么意思呢？
            //我们画出来的点不一定是一个点，而是一个很小的圆形
            //所以我们就用画圆形的方法
            //上面迭代器把ArrayList集合中的Object类型元素迭代出来时已经强转成了Point类型
            //我们就用point提供的坐标来传入画圆形的方法中，这样一个一个的小圆形就被画出来了，也就是我们看到的点

        }

    }

    //添加一个点到界面上面，how？用paint把点画上去
    //怎么画？先把集合中的点迭代出来再画
    //但是Frame窗口怎么才能接收到我们传进来的点呢？（传进Frame中才能执行画笔方法画画）
    //通过一个中间商,自己写一个add点的方法
    public void addPoint(Point point){
        points.add(point);
    }

    private class MyMouseListener extends MouseAdapter{
        //如果直接实现MouseListener接口则需要重写一大堆抽象方法
        //根据适配器模式，我们可以继承他的子类，这样可以不用重写那么多方法
        //鼠标有三个动作：按下，按住不动，弹起

        @Override
        public void mousePressed(MouseEvent e) {
            MyFrame myFrame =(MyFrame) e.getSource();
            //这里谁调用的它就是e.getSource()就返回谁
            //所以我们这里用多态创建一个引用变量来接收它
            //注意：getSource()方法返回的是一个Object类型的对象，所以需要使用多态，同时还要向下转型
            //这里我们点击的时候就会在我们传进来的窗口的界面上产生一个点
            //现在我们要做的就是把这个被添加到窗口中的点显现出来
            //这个点要被画出来，所以现在我们要把思路转到画笔这里来
            //这个点就是鼠标的点

            //通过这个方法把点传到窗口的存点集合里面去
            myFrame.addPoint(new Point(e.getX(),e.getY()));
            //每次点击鼠标都需要重画一遍，不然往集合中画一次点程序就结束了
            myFrame.repaint();
            //这个方法可以在画完点后刷新一次
        }
    }
}

    public class JFrameDemo {

    public void init(){

    //游戏的面板
public class GamePanel extends JPanel implements KeyListener, ActionListener {
    //蛇会移动，还有坐标，这个时候我们需要去定义一些关于蛇的数据结构
    int length;  //蛇的长度
    int[] snakeX = new int[600];//蛇的x坐标 25*24
    int[] snakeY = new int[500];//蛇的x坐标 25*20
    String fx; //初始方向向右
    //游戏当前的状态：开始，停止
    boolean isStart = false; //默认游戏不开始

    boolean isFail = false; //游戏失败状态

    //食物的坐标
    int foodX;
    int foodY;
    //还需要一个随机数种子，因为食物的坐标是随机出现的
    Random random = new Random(); //默认的随机数种子

    int score; //成绩


    //定时器，只需要实现ActionListener接口,重写一个定时的方法就可以了，定时器是一个Timer类，new一个就可以了
    Timer timer = new Timer(100, this);
    //解释一下参数，后面的this代表被定时器监听的对象
    //前面的delay代表多少毫秒刷新一次，100毫秒就是一秒刷新十次
    //不要设置的太快，否则小蛇就直接起飞了，跟QQ堂外挂一样

    public GamePanel() {
        init();
        //获得焦点和键盘事件
        this.setFocusable(true); //获得焦点事件
        //焦点是什么意思，，昨天将文本框时解释过
        //你在一个文本框内操作的时候它会聚焦在这个文本框上，
        // 你鼠标一旦离开文本框点击了其它东西那么它便不再聚焦在这里面

        //获取键盘事件
        this.addKeyListener(this);
        //由于键盘监听事件写在当前这个类中，所以直接this
        //如果是写在其他类或内部类中，则需要去new一个监听器

        timer.start();
        //游戏一开始就启动定时器，相当于开始执行小蛇移动的监听器方法
    }

    //初始化方法
    public void init() {
        length = 3;
        snakeX[0] = 100;
        snakeY[0] = 100; //头的初始坐标
        snakeX[1] = 75;
        snakeY[1] = 100; //第一个身体的初始坐标
        snakeX[2] = 50;
        snakeY[2] = 100; //第二个身体的初始坐标
        fx = "R"; //初始方向向右
        //养成一个习惯，数据的初始化都放在初始化方法里，默认的变量（定义）放在最上面，这样分开比较好

        //初始化食物坐标数据
        foodX = 25 + 25*random.nextInt(34); //之前计算过X总共就34个格子，这个代表的是不超过34的整数
        foodY = 75 + 25*random.nextInt(24); //Y总共有24个格子

        //初始化成绩
        score = 0;
    }

    //画板： 画界面，画蛇
    //绘制面板，我们游戏中的所有东西都使用这个画笔来画
    @Override
    protected void paintComponent(Graphics g) {
        super.paintComponent(g);//清屏，不用的话也可以，但是会出现闪屏

        //绘制静态的面板,注意一切绘画操作都是在这个重写的绘制组件方法里面完成的
        Data.header.paintIcon(this, g, 25, 11);//绘制头部广告栏
        this.setBackground(Color.WHITE);//设置面板的背景颜色
        //在下面绘制一个黑色的长方形矩形作为游戏界面
        g.fillRect(25, 75, 850, 600);

        //画积分
        g.setColor(Color.WHITE);
        g.setFont(new Font("微软雅黑",Font.BOLD,18));
        g.drawString("长度" + length,750,35);
        g.drawString("分数" + score,750,50);

        //画食物
        Data.food.paintIcon(this,g,foodX,foodY);


        //把小蛇画上去
        //Icon类的变量可以调用一个叫做paintIcon的方法，第一个参数穿传要被画上去的面板，
        //后面一个参数传画笔，最后两个参数传坐标
        if (fx.equals("R")) {
            Data.right.paintIcon(this, g, snakeX[0], snakeY[0]); //蛇头初始化向右，需要通过方向来判断
        } else if (fx.equals("L")) {
            Data.left.paintIcon(this, g, snakeX[0], snakeY[0]);
        } else if (fx.equals("U")) {
            Data.up.paintIcon(this, g, snakeX[0], snakeY[0]);
        } else if (fx.equals("D")) {
            Data.down.paintIcon(this, g, snakeX[0], snakeY[0]);
        }


        for (int i = 1; i < length; i++) {
            Data.body.paintIcon(this, g, snakeX[i], snakeY[i]);
        }
        //用上面for循环的方式则下面两行就用不上了
        //Data.body.paintIcon(this,g,snakeX[1],snakeY[1]); //蛇的第一个身体
        //Data.body.paintIcon(this,g,snakeX[2],snakeY[2]); //蛇的第二个身体


        //游戏状态
        if (isStart == false) {
            g.setColor(Color.WHITE); //将画文字的画笔设置为白色
            g.setFont(new Font("微软雅黑", Font.BOLD, 40)); //设置字体的方法
            g.drawString("按下空格开始游戏！", 300, 300); //画文字的方法
        }
        if (isFail) {
            g.setColor(Color.RED);
            g.setFont(new Font("微软雅黑",Font.BOLD,40));
            g.drawString("游戏失败，按下空格重新开始",200,300);
        }
    }



    //键盘监听器
    //最暴力的方式，直接让当前面板实现键盘监听器接口，不用单独再写一个类，连内部类都不用写
    //键盘监听事件
    @Override
    public void keyPressed(KeyEvent e) {
        //首先监听空格，要获得当前键盘的keyCode
        int keyCode = e.getKeyCode(); //获得键盘按键是哪一个
        if (keyCode == KeyEvent.VK_SPACE) {
            //如果按下的是空格键
            if(isFail){
                //重新开始
                isFail = false;
                init();
            } else{
                isStart = !isStart; //取反操作
            }
            repaint(); //就像那天的画小圆点一样，监测完了键盘监听事件之后要repaint()
            //repaint()完了之后相当于原本屏幕上面的那行字就消失了，因为panel进行了repaint
        }
        //设置键盘监听控制小蛇上下左右移动
        if (keyCode == KeyEvent.VK_UP) {
            fx = "U";
        } else if (keyCode == KeyEvent.VK_DOWN) {
            fx = "D";
        } else if (keyCode == KeyEvent.VK_LEFT) {
            fx = "L";
        } else if (keyCode == KeyEvent.VK_RIGHT) {
            fx = "R";
        }
    }


    //下面两个用不上的抽象方法放着就行了
    @Override
    public void keyTyped(KeyEvent e) {
    }

    @Override
    public void keyReleased(KeyEvent e) {
    }

    @Override
    public void actionPerformed(ActionEvent e) { //定时的事件监听方法
        //事件监听需要通过固定的时间来刷新，比如1秒10次（频率可以自己调）
        if (isStart == true && isFail == false) { //如果游戏是开始状态，就让小蛇动起来

            //总共有两个监听，一个键盘监听，一个事件监听，吃食物放在事件监听里面
            if(snakeX[0] == foodX && snakeY[0] == foodY){
                length++; //长度加一再重新生成食物即可
                score = score + 10; //每吃一个分数加10
                foodX = 25 + 25*random.nextInt(34);
                foodY = 75 + 25*random.nextInt(24);
            }


            //动就是让小蛇的坐标一直加，先默认右移
            for (int i = length - 1; i > 0; i--) {
                snakeX[i] = snakeX[i - 1]; //向前移动一节
                snakeY[i] = snakeY[i - 1]; //向前移动一节
                //最开始头是在右边的，但是头是数组最开始的那一个
                //所以头动了就让跟着的一节身体移动到头的位置
                //然后然后让接下来的第二届身体移动到第一节身体的位置
                //第一节身体变到第0节身体，第二节身体变到第一节身体
            }
            //控制走向（身体不用控制，后面一节自动到前面一节，只用控制脑袋的走向即可
            if (fx.equals("R")) {
                snakeX[0] = snakeX[0] + 25; //头部移动
                if (snakeX[0] > 850) {
                    snakeX[0] = 25;
                } //边界判断
            } else if (fx.equals("L")) {
                snakeX[0] = snakeX[0] - 25;
                if (snakeX[0] < 25) {
                    snakeX[0] = 850;
                } //边界判断
            } else if(fx.equals("U")) {
                snakeY[0] = snakeY[0] -25;
                if(snakeY[0] < 75){
                    snakeY[0] = 650;
                }
            } else if(fx.equals("D")) {
                snakeY[0] = snakeY[0] +25;
                if(snakeY[0] > 650){
                    snakeY[0] = 75;
                }
            }

            //失败判定，撞到自己就算失败
            for(int i = 1; i < length; i++) {
                if(snakeX[0] == snakeX[i] && snakeY[0] == snakeY[i]) {
                    isFail = true;
                }
            }
            repaint(); //变完了就需要让它重画一下
            timer.start(); //定时器开启
        }

    }
}
